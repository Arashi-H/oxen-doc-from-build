92  cn_heavy_hash() : borrowed_pad(false)

100  static cn_heavy_hash_v1 make_borrowed(cn_heavy_hash_v2& t)
101  {
102  return {t.lpad.as_void(), t.spad.as_void()};
103  }

105  // Disable copy/move ctors; copying, in particular, is going to be really inefficient and we

117  void hash(const void* in, size_t len, void* out, bool prehashed=false)
118  {

133 private:

135  friend cn_heavy_hash_v1;
136  friend cn_heavy_hash_v2;
137 
138  // Constructor enabling v1 hash to borrow v2's buffer
139  cn_heavy_hash(void* lptr, void* sptr)
140  {
141  lpad.set(lptr);
142  spad.set(sptr);
143  borrowed_pad = true;
144  }

146  inline void free_mem()
147  {
148  if(!borrowed_pad)
149  {
150  if(lpad.as_void() != nullptr)
151  boost::alignment::aligned_free(lpad.as_void());
152  if(lpad.as_void() != nullptr)
153  boost::alignment::aligned_free(spad.as_void());
154  }
155 
156  lpad.set(nullptr);
157  spad.set(nullptr);
158  }
