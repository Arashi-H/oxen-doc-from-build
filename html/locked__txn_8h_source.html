37  // This class is meant to create a batch when none currently exists.
38  // If a batch exists, it can't be from another thread, since we can
39  // only be called with the txpool lock taken, and it is held during
40  // the whole prepare/handle/cleanup incoming block sequence.
41  class LockedTXN {
42  public:
43  LockedTXN(Blockchain &b): m_db{b.get_db()} {
44  m_batch = m_db.batch_start();
45  }
46  LockedTXN(const LockedTXN &) = delete;
47  LockedTXN &operator=(const LockedTXN &) = delete;
48  LockedTXN(LockedTXN &&o) : m_db{o.m_db}, m_batch{o.m_batch} { o.m_batch = false; }
49  LockedTXN &operator=(LockedTXN &&) = delete;
50 
51  void commit() { try { if (m_batch) { m_db.batch_stop(); m_batch = false; } } catch (const std::exception &e) { log::warning(globallogcat, "LockedTXN::commit filtering exception: {}", e.what()); } }
52  void abort() { try { if (m_batch) { m_db.batch_abort(); m_batch = false; } } catch (const std::exception &e) { log::warning(globallogcat, "LockedTXN::abort filtering exception: {}", e.what()); } }
53  ~LockedTXN() { this->abort(); }
54  private:
55  BlockchainDB &m_db;
56  bool m_batch;
57  };
