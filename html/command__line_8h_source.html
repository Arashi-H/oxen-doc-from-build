151  template<typename T>
152  boost::program_options::typed_value<T, char>* make_semantic(const arg_descriptor<T, false, true>& arg)
153  {
154  auto semantic = boost::program_options::value<T>();
155  if (!arg.not_use_default) {
156  auto default_display = "{}, {} if '{}'"_format(
157  arg_stringify(arg.depf(false, true, arg.default_value)),
158  arg_stringify(arg.depf(true, true, arg.default_value)),
159  arg.ref.name);
160  semantic->default_value(arg.depf(arg.ref.default_value, true, arg.default_value), default_display);
161  }
162  return semantic;
163  }

165  template<typename T, int NUM_DEPS>
166  boost::program_options::typed_value<T, char>* make_semantic(const arg_descriptor<T, false, true, NUM_DEPS>& arg)
167  {
168  auto semantic = boost::program_options::value<T>();
169  if (!arg.not_use_default) {
170  std::array<bool, NUM_DEPS> depval;
171  depval.fill(false);
172  auto default_display = arg_stringify(arg.depf(depval, true, arg.default_value));
173  for (size_t i = 0; i < depval.size(); ++i)
174  {
175  depval.fill(false);
176  depval[i] = true;
177  fmt::format_to(std::back_inserter(default_display),
178  ", {} if '{}'",
179  arg_stringify(arg.depf(depval, true, arg.default_value)),
180  arg.ref[i]->name);
181  }
182  for (size_t i = 0; i < depval.size(); ++i)
183  depval[i] = arg.ref[i]->default_value;
184  semantic->default_value(arg.depf(depval, true, arg.default_value), default_display);
185  }
186  return semantic;
187  }
