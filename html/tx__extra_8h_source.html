182  template <class Archive>
183  void serialize_value(Archive& ar, tx_extra_padding& pad)
184  {
185  size_t remaining;
186  if constexpr (Archive::is_deserializer)
187  remaining = ar.remaining_bytes();
188  else if (pad.size <= 1)
189  return;
190  else
191  remaining = pad.size - 1; // - 1 here (and just below) because we consider the 0x00 variant tag part of the padding
192 
193  if (remaining > TX_EXTRA_PADDING_MAX_COUNT - 1) // - 1 as above.
194  throw std::invalid_argument{"tx_extra_padding size is larger than maximum allowed"};
195 
196  char buf[TX_EXTRA_PADDING_MAX_COUNT - 1] = {};
197  ar.serialize_blob(buf, remaining);
198 
199  if (Archive::is_deserializer)
200  {
201  if (std::string_view{buf, remaining}.find_first_not_of('\0') != std::string::npos)
202  throw std::invalid_argument{"Invalid non-0 padding byte"};
203  pad.size = remaining + 1;
204  }
205  }

404  template <class Archive>
405  void serialize_value(Archive& ar) {
406  // Retrofit a field version in here. Prior to adding reason fields, the first value (in
407  // binary serialization) was the `state` enum, which had a maximum acceptable value of 3: so
408  // if we get >= 4, that's a version, and otherwise we're implicitly version 0 (and there is no
409  // version 1-3).
410  if (Archive::is_serializer && version >= version_t::v4_reasons) {
411  field_varint(ar, "version", version);
412  } else if constexpr (Archive::is_deserializer) {
413  uint8_t ver;
414  field_varint(ar, "version", ver, [](auto v) { return v <= 4; });
415  if (ver < 4) { // Old record, so the "version" we read is actually the state value
416  version = version_t::v0;
417  state = static_cast<service_nodes::new_state>(ver);
418  } else {
419  version = static_cast<version_t>(ver);
420  }
421  }
422  if (Archive::is_serializer || version >= version_t::v4_reasons) {
423  field_varint(ar, "state", state, [](auto s) { return s < service_nodes::new_state::_count; });
424  }
425 
426  field_varint(ar, "block_height", block_height);
427  field_varint(ar, "service_node_index", service_node_index);
428  field(ar, "votes", votes);
429  if (version >= version_t::v4_reasons)
430  {
431  field_varint(ar, "reason_consensus_all", reason_consensus_all);
432  field_varint(ar, "reason_consensus_any", reason_consensus_any);
433  }
434  }
435  };
