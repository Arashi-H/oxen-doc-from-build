132  // Steps a statement to completion that is expected to return at most one row, optionally binding
133  // values into it (if provided). Returns a filled out optional<T> (or optional<std::tuple<T...>>)
134  // if a row was retrieved, otherwise a nullopt. Throws if more than one row is retrieved.
135  template <typename... T, typename... Args>
136  std::optional<type_or_tuple<T...>>
137  exec_and_maybe_get(SQLite::Statement& st, const Args&... bind)
138  {
139  bind_oneshot(st, bind...);
140  std::optional<type_or_tuple<T...>> result;
141  while (st.executeStep())
142  {
143  if (result)
144  {
145  log::error(sqlitedb_logcat, "Expected single-row result, got multiple rows from {}", st.getQuery());
146  throw std::runtime_error{"DB error: expected single-row result, got multiple rows"};
147  }
148  result = get<T...>(st);
149  }
150  return result;
151  }

153  // Executes a statement to completion that is expected to return exactly one row, optionally
154  // binding values into it (if provided). Returns a T or std::tuple<T...> (depending on whether or
155  // not more than one T is provided) for the row. Throws an exception if no rows or more than one
156  // row are returned.
157  template <typename... T, typename... Args>
158  type_or_tuple<T...>
159  exec_and_get(SQLite::Statement& st, const Args&... bind)
160  {

166  }
