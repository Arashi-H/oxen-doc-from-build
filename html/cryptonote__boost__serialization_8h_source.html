171  inline void serialize(Archive &a, cryptonote::transaction_prefix &x, const boost::serialization::version_type ver)
172  {
173  a & x.version;
174  if (x.version >= cryptonote::txversion::v3_per_output_unlock_times)
175  {
176  a & x.output_unlock_times;
177  if (x.version == cryptonote::txversion::v3_per_output_unlock_times) {
178  bool is_deregister = x.type == cryptonote::txtype::state_change;
179  a & is_deregister;
180  x.type = is_deregister ? cryptonote::txtype::state_change : cryptonote::txtype::standard;
181  }
182  }
183  a & x.unlock_time;
184  a & x.vin;
185  a & x.vout;
186  a & x.extra;
187  if (x.version >= cryptonote::txversion::v4_tx_types)
188  a & x.type;

312  template <class Archive>
313  inline typename std::enable_if<Archive::is_loading::value, void>::type serializeOutPk(Archive &a, rct::ctkeyV &outPk_, const boost::serialization::version_type ver)
314  {
315  rct::keyV outPk;
316  a & outPk;
317  outPk_.resize(outPk.size());
318  for (size_t n = 0; n < outPk_.size(); ++n)
319  {
320  outPk_[n].dest = rct::identity();
321  outPk_[n].mask = outPk[n];
322  }
323  }

334  template <class Archive>
335  inline void serialize(Archive &a, rct::rctSigBase &x, const boost::serialization::version_type ver)
336  {
337  a & x.type;
338  if (x.type == rct::RCTType::Null)
339  return;
340  if (!tools::equals_any(x.type, rct::RCTType::Full, rct::RCTType::Simple, rct::RCTType::Bulletproof, rct::RCTType::Bulletproof2, rct::RCTType::CLSAG))
341  throw boost::archive::archive_exception(boost::archive::archive_exception::other_exception, "Unsupported rct type");
342  // a & x.message; message is not serialized, as it can be reconstructed from the tx data
343  // a & x.mixRing; mixRing is not serialized, as it can be reconstructed from the offsets
344  if (x.type == rct::RCTType::Simple) // moved to prunable with bulletproofs
345  a & x.pseudoOuts;
346  a & x.ecdhInfo;
347  serializeOutPk(a, x.outPk, ver);
348  a & x.txnFee;
349  }
