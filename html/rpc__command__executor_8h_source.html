73  template <typename RPC, std::enable_if_t<std::is_base_of_v<cryptonote::rpc::RPC_COMMAND, RPC> && cryptonote::rpc::FIXME_has_nested_response_v<RPC>, int> = 0>
74  bool invoke(typename RPC::request&& req, typename RPC::response& res, const std::string& error, bool check_status_ok = true)
75  {
76  try {
77  if (auto* rpc_client = std::get_if<cryptonote::rpc::http_client>(&m_rpc)) {
78  res = rpc_client->json_rpc<RPC>(RPC::names()[0], req);
79  } else {
80  throw std::runtime_error{"fixme"};
81  }
82  if (!check_status_ok || res.status == cryptonote::rpc::STATUS_OK)
83  return true;
84  } catch (const std::exception& e) {
85  if (!error.empty())
86  tools::fail_msg_writer("{}: {}", error, e.what());
87  return false;
88  } catch (...) {}
89  if (!error.empty())
90  tools::fail_msg_writer(error);
91  return false;
92  }

122  template <typename RPC, std::enable_if_t<std::is_base_of_v<cryptonote::rpc::RPC_COMMAND, RPC> && !cryptonote::rpc::FIXME_has_nested_response_v<RPC>, int> = 0>
123  nlohmann::json invoke(std::optional<nlohmann::json> params = std::nullopt, bool check_status_ok = true)

128  // Invokes a simple RPC method that doesn't take any arguments and for which we don't care about
129  // the return value beyond the "status": "OK" field. Returns true (and prints a success message)
130  // on success, false (with a failure message printed) on failure.
131  template <typename RPC, std::enable_if_t<std::is_base_of_v<cryptonote::rpc::RPC_COMMAND, RPC> && !cryptonote::rpc::FIXME_has_nested_response_v<RPC>, int> = 0>
132  bool invoke_simple(std::string_view error_prefix, std::string_view success_msg) {
133  if (!try_running([this] { return invoke<RPC>(); }, error_prefix))
134  return false;
135 
136  tools::success_msg_writer(success_msg);
137  return true;
138  }
