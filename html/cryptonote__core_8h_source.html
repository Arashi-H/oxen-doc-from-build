78  // so that core doesn't need to link against cryptonote_protocol (plus everything it depends on).

80  // Initializes quorumnet state (for service nodes only). This is called after the OxenMQ object
81  // has been set up but before it starts listening. Return an opaque pointer (void *) that gets
82  // passed into all the other callbacks below so that the callbacks can recast it into whatever it
83  // should be.
84  using quorumnet_new_proc = void *(core &core);
85  // Initializes quorumnet; unlike `quorumnet_new_proc` this needs to be called for all nodes, not
86  // just service nodes. The second argument should be the `quorumnet_new` return value if a
87  // service node, nullptr if not.
88  using quorumnet_init_proc = void (core &core, void *self);
89  // Destroys the quorumnet state; called on shutdown *after* the OxenMQ object has been destroyed.
90  // Should destroy the state object and set the pointer reference to nullptr.
91  using quorumnet_delete_proc = void (void *&self);
92  // Relays votes via quorumnet.
93  using quorumnet_relay_obligation_votes_proc = void (void *self, const std::vector<service_nodes::quorum_vote_t> &votes);
94  // Sends a blink tx to the current blink quorum, returns a future that can be used to wait for the
95  // result.
96  using quorumnet_send_blink_proc = std::future<std::pair<blink_result, std::string>> (core& core, const std::string& tx_blob);

98  // Relay a Pulse message to members specified in the quorum excluding the originating message owner.
