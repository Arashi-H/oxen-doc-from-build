81  // Maximum number of bytes of a varint-encoded integer of the given type.
82  template <typename T>
83  constexpr size_t VARINT_MAX_LENGTH = (sizeof(T) * 8 + 6) / 7; // i.e. integer ceil(bits / 7)

89  template <typename OutputIt, typename T, std::enable_if_t<std::is_integral_v<T> && std::is_unsigned_v<T>, int> = 0>
90  void write_varint(OutputIt&& it, T i) {
91 #ifndef NDEBUG
92  size_t count = 0;
93 #endif
94  while (i > 0b0111'1111) {
95  // Write the lower 7 bits with the 8th continuation bit set:
96  assert(++count < VARINT_MAX_LENGTH<T>);
97  *it++ = static_cast<char>((i & 0b0111'1111) | 0b1000'0000);
98  i >>= 7;
99  }
100  // Last byte (7 bits or less, no continuation bit)
101  assert(++count <= VARINT_MAX_LENGTH<T>);
102  *it++ = static_cast<char>(i);
103  }

119  template <typename It, typename EndIt, typename T, std::enable_if_t<std::is_integral_v<T> && std::is_unsigned_v<T>, int> = 0>
120  int read_varint(It&& it, const EndIt& end, T& write) {
121  constexpr size_t bits = sizeof(T) * 8;
122 
123  bool more = true;
124  int read = 0;
125  write = 0;
126  for (size_t shift = 0; more && it != end; shift += 7)
127  {
128  auto byte = static_cast<unsigned char>(*it++);
129  ++read;
130 
131  // If byte is all 0s and this isn't the first byte then something is wrong: we have a final
132  // byte containing nothing significant, but we should never have produced that.
133  if (byte == 0 && shift)
134  return EVARINT_REPRESENT;
135 
136  // If we have <= 7 bits of space remaining then the value must fit and must not have a continuation bit
137  if (size_t bits_avail = bits - shift; bits_avail <= 7 && byte >= 1 << bits_avail)
138  return EVARINT_OVERFLOW;
139 
140  more = byte & 0b1000'0000; // continuation bit
141 
142  write |= static_cast<T>(byte & 0b0111'1111) << shift; // 7-bit value
143  }
144 
145  return more ? EVARINT_TRUNCATED : read;
146  }
