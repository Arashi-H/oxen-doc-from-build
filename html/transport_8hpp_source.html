162  // Flexible json serialization. HTTP client tailored for bridge API
163  template<class t_req, class t_res>
164  bool invoke_bridge_http(std::string_view uri, const t_req & request, t_res & result)
165  {
166  std::string req;
167  t_serialize(request, req);
168 
169  std::string res;
170  OXEN_DEFER { if (!res.empty()) memwipe(res.data(), res.size()); };
171 
172  try {
173  res = post_json(uri, std::move(req));
174  } catch (const std::exception& e) {
175  MERROR(e.what() << " while requesting " << uri);
176  }
177 
178  return t_deserialize(res, result);
179  }

366  template<class t_message=google::protobuf::Message>
367  std::shared_ptr<t_message>
368  exchange_message(Transport & transport, const google::protobuf::Message & req,
369  std::optional<messages::MessageType> resp_type = std::nullopt)
370  {
371  // Require strictly protocol buffers response in the template.
372  static_assert(std::is_base_of_v<google::protobuf::Message, t_message>);
373 
374  // Write the request
375  transport.write(req);
376 
377  // Read the response
378  std::shared_ptr<google::protobuf::Message> msg_resp;
379  hw::trezor::messages::MessageType msg_resp_type;
380  transport.read(msg_resp, &msg_resp_type);
381 
382  // Determine type of expected message response
383  messages::MessageType required_type = resp_type ? *resp_type : MessageMapper::get_message_wire_number<t_message>();
384 
385  if (msg_resp_type == required_type) {
386  return message_ptr_retype<t_message>(msg_resp);
387  } else if (msg_resp_type == messages::MessageType_Failure){
388  throw_failure_exception(dynamic_cast<messages::common::Failure*>(msg_resp.get()));
389  } else {
390  throw exc::UnexpectedMessageException(msg_resp_type, msg_resp);
391  }
392  }
