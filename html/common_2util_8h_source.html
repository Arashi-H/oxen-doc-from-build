108  // Copy a class byte-for-byte (but only if it is standard layout and has byte alignment)
109  template <typename T, std::enable_if_t<std::is_class<T>::value && !is_crypto_bytes_derived<T>, int> = 0>
110  void memcpy_one(char*& dest, const T& t) {
111  // We don't *actually* require byte alignment here but it's quite possibly an error (i.e.
112  // passing in a type containing integer members) so disallow it.
113  static_assert(std::is_trivially_copyable<T>::value && alignof(T) == 1, "memcpy_le() may only be used on simple (1-byte alignment) struct types");
114  std::memcpy(dest, &t, sizeof(T));
115  dest += sizeof(T);
116  }

133  // Does a memcpy of one or more values into a char array; for any given values that are basic
134  // integer types the value is first converted from native to little-endian representation (if
135  // necessary). Non-integer types with alignment of 1 (typically meaning structs containing only
136  // char, bools, and arrays of those) and fixed-size arrays of the above (including string
137  // literals) are also permitted; more complex types are not.
138  //
139  // The 1-byte alignment is here to protect you: if you have a larger alignment that usually means
140  // you have a contained type with a larger alignment, which is probably an integer.
141  template <typename... T>
142  auto memcpy_le(const T &...t) {
143  std::array<char, (0 + ... + detail::memcpy_size<T>)> r;
144  char* dest = r.data();
145  (..., detail::memcpy_one(dest, t));
146  return r;
147  }
