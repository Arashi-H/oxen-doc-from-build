62  template <class Archive, class ver_type>
63  inline void serialize(Archive &a, epee::net_utils::network_address& na, const ver_type ver)
64  {
65  static constexpr const typename Archive::is_saving is_saving{};
66 
67  uint8_t type;
68  if (is_saving)
69  type = uint8_t(na.get_type_id());
70  a & type;
71  switch (epee::net_utils::address_type(type))
72  {
73  case epee::net_utils::ipv4_network_address::get_type_id():
74  do_serialize<epee::net_utils::ipv4_network_address>(is_saving, a, na);
75  break;
76  case epee::net_utils::ipv6_network_address::get_type_id():
77  do_serialize<epee::net_utils::ipv6_network_address>(is_saving, a, na);
78  break;
79  case net::tor_address::get_type_id():
80  do_serialize<net::tor_address>(is_saving, a, na);
81  break;
82  case net::i2p_address::get_type_id():
83  do_serialize<net::i2p_address>(is_saving, a, na);
84  break;
85  case epee::net_utils::address_type::invalid:
86  default:
87  throw std::runtime_error("Unsupported network address type");
88  }
89  }

159  template <class Archive, class ver_type>
160  inline void load(Archive& a, net::tor_address& na, const ver_type)
161  {
162  uint16_t port = 0;
163  uint8_t length = 0;
164  a & port;
165  a & length;
166 
167  const size_t buffer_size = net::tor_address::buffer_size();
168  if (length > buffer_size)
169  MONERO_THROW(net::error::invalid_tor_address, "Tor address too long");
170 
171  char host[buffer_size] = {0};
172  a.load_binary(host, length);
173  host[sizeof(host) - 1] = 0;
174 
175  if (std::strcmp(host, net::tor_address::unknown_str()) == 0)
176  na = net::tor_address::unknown();
177  else
178  na = MONERO_UNWRAP(net::tor_address::make(host, port));
179  }

216  inline void serialize(Archive &a, nodetool::peerlist_entry& pl, const ver_type ver)
217  {
218  a & pl.adr;
219  a & pl.id;
220  a & pl.last_seen;
221  if (ver < 1)

226  }

229  {

231  }
232  if (ver < 2)

234  return;
