64  {

67  fs::ifstream data_file{file_path, std::ios_base::binary};
68  if (data_file.fail())

71  {
72  // first try reading in portable mode
73  boost::archive::portable_binary_iarchive{data_file} >> obj;
74  }
75  catch(...)
76  {
77  // if failed, try reading in unportable mode
78  auto unportable = file_path;
79  unportable += ".unportable";

81  data_file.close();

83  if (data_file.fail())
