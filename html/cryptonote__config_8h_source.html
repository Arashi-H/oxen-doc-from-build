250 constexpr network_type network_type_from_string(std::string_view s)
251 {
252  if (s == "mainnet")
253  return network_type::MAINNET;
254  if (s == "testnet")
255  return network_type::TESTNET;
256  if (s == "devnet")
257  return network_type::DEVNET;
258  if (s == "fakechain")
259  return network_type::FAKECHAIN;
260 
261  return network_type::UNDEFINED;
262 }
263 
264 constexpr std::string_view network_type_to_string(network_type t)
265 {
266  switch (t) {
267  case network_type::MAINNET:
268  return "mainnet";
269  case network_type::TESTNET:
270  return "testnet";
271  case network_type::DEVNET:
272  return "devnet";
273  case network_type::FAKECHAIN:
274  return "fakechain";
275  default:
276  return "undefined";
277  }
278  return "undefined";
279 }

304  inline constexpr uint64_t DIFFICULTY_WINDOW = 59;
305  inline constexpr uint64_t DIFFICULTY_BLOCKS_COUNT(bool before_hf16)
306  {
307  // NOTE: We used to have a different setup here where,
308  // DIFFICULTY_WINDOW = 60
309  // DIFFICULTY_BLOCKS_COUNT = 61
310  // next_difficulty_v2's N = DIFFICULTY_WINDOW - 1
311  //
312  // And we resized timestamps/difficulties to (N+1) (chopping off the latest timestamp).
313  //
314  // Now we re-adjust DIFFICULTY_WINDOW to 59. To preserve the old behaviour we
315  // add +2. After HF16 we avoid trimming the top block and just add +1.
316  //
317  // Ideally, we just set DIFFICULTY_BLOCKS_COUNT to DIFFICULTY_WINDOW
318  // + 1 for before and after HF16 (having one unified constant) but this
319  // requires some more investigation to get it working with pre HF16 blocks and
320  // alt chain code without bugs.
321  uint64_t result = (before_hf16) ? DIFFICULTY_WINDOW + 2 : DIFFICULTY_WINDOW + 1;
322  return result;
323  }
