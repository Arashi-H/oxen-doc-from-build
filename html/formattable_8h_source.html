10  // Types can opt-in to being formattable as a string by specializing this to true. Such a type
11  // must have one of:
12  //
13  // - a `to_string()` method on the type; when formatted we will call `val.to_string()` to format
14  // it as a string.
15  // - a `to_string(val)` function in the same namespace as the type; we will call it to format it
16  // as a string.
17  //
18  // The function should return something string-like (string, string_view, const char*).
19  //
20  // For instance to opt-in MyType for such string formatting, use:
21  //
22  // template <> inline constexpr bool formattable::via_to_string<MyType> = true;
23  //
24  // You can also partially specialize; for instance to make all derived classes of a common base
25  // type formattable via to_string you could do:
26  //
27  // template <T> inline constexpr bool formattable::via_to_string<T,
28  // std::enable_if_t<std::is_base_of_v<MyBaseType, T>>> = true;
29  //
30  template <typename T, typename SFINAE = void>
31  constexpr bool via_to_string = false;
