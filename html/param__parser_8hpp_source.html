117  // Consumes the next value from the dict consumer into `val`
118  template <typename BTConsumer, typename T,
119  std::enable_if_t<
120  std::is_same_v<BTConsumer, bt_dict_consumer>
121  || std::is_same_v<BTConsumer, bt_list_consumer>,
122  int> = 0>
123  void load_value(BTConsumer& c, T& val) {
124  if constexpr (std::is_integral_v<T>)
125  val = c.template consume_integer<T>();
126  else if constexpr (std::is_same_v<T, std::string> || std::is_same_v<T, std::string_view>)
127  val = c.consume_string_view();
128  else if constexpr (is_binary_parameter<T>)
129  load_binary_parameter(c.consume_string_view(), true /*allow raw*/, val);
130  else if constexpr (is_expandable_list<T>) {
131  auto lc = c.consume_list_consumer();
132  val.clear();
133  while (!lc.is_finished())
134  load_value(lc, val.emplace_back());
135  }
136  else if constexpr (is_tuple_like<T>) {
137  auto lc = c.consume_list_consumer();
138  load_tuple_values(lc, val, std::make_index_sequence<std::tuple_size_v<T>>{});
139  }
140  else
141  static_assert(std::is_same_v<T, void>, "Unsupported load_value type");
142  }
143  // Copies the next value from the json range into `val`, and advances the iterator. Throws
144  // on unconvertible values.
145  template <typename T>
146  void load_value(json_range& r, T& val) {
147  auto& key = r.first.key();
148  auto& e = *r.first;
149  if constexpr (std::is_same_v<T, bool>) {
150  if (e.is_boolean())
151  val = e.get<bool>();
152  else if (e.is_number_unsigned()) {
153  // Also accept 0 or 1 for bools (mainly to be compatible with bt-encoding which doesn't
154  // have a distinct bool type).
155  auto b = e.get<uint64_t>();
156  if (b <= 1)
157  val = b;
158  else
159  throw std::domain_error{"Invalid value for '" + key + "': expected boolean"};
160  } else {
161  throw std::domain_error{"Invalid value for '" + key + "': expected boolean"};
162  }
163  } else if constexpr (std::is_unsigned_v<T>) {
164  if (!e.is_number_unsigned())
165  throw std::domain_error{"Invalid value for '" + key + "': non-negative value required"};
166  auto i = e.get<uint64_t>();
167  if (sizeof(T) < sizeof(uint64_t) && i > std::numeric_limits<T>::max())
168  throw std::domain_error{"Invalid value for '" + key + "': value too large"};
169  val = i;
170  } else if constexpr (std::is_integral_v<T>) {
171  if (!e.is_number_integer())
172  throw std::domain_error{"Invalid value for '" + key + "': value is not an integer"};
173  auto i = e.get<int64_t>();
174  if (sizeof(T) < sizeof(int64_t)) {
175  if (i < std::numeric_limits<T>::lowest())
176  throw std::domain_error{"Invalid value for '" + key + "': negative value magnitude is too large"};
177  else if (i > std::numeric_limits<T>::max())
178  throw std::domain_error{"Invalid value for '" + key + "': value is too large"};
179  }
180  val = i;
181  } else if constexpr (std::is_same_v<T, std::string> || std::is_same_v<T, std::string_view>) {
182  val = e.get<std::string_view>();
183  } else if constexpr (is_binary_parameter<T> || is_expandable_list<T> || is_tuple_like<T>) {
184  try { e.get_to(val); }
185  catch (const std::exception& e) { throw std::domain_error{"Invalid values in '" + key + "'"}; }
186  } else {
187  static_assert(std::is_same_v<T, void>, "Unsupported load type");
188  }
189  ++r.first;
190  }

197  // Takes a json object iterator or bt_dict_consumer and loads the current value at the iterator.
198  // This calls itself recursively, if needed, to unwrap optional/required/ignore_empty_string
199  // wrappers.
200  template <typename In, typename T>
201  void load_curr_value(In& in, T& val) {
202  if constexpr (is_required_wrapper<T>) {
203  load_curr_value(in, val.value);

257  }
