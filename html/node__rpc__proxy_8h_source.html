73  // Invokes an JSON RPC request and checks it for errors, include a check that the response
74  // `.status` value is equal to rpc::STATUS_OK. Returns the response on success, logs and throws
75  // on error.
76  template <typename RPC>
77  typename RPC::response invoke_json_rpc(const typename RPC::request& req) const
78  {
79  typename RPC::response result;
80  try {
81  result = m_http_client.json_rpc<RPC>(RPC::names().front(), req);
82  } catch (const std::exception& e) {
83  log::error(globallogcat, e.what());
84  throw;
85  }
86  if (result.status != cryptonote::rpc::STATUS_OK) {
87  std::string error = "Request for " + std::string{RPC::names().front()} + " failed: " + (result.status == cryptonote::rpc::STATUS_BUSY ? "daemon is busy" : result.status);
88  log::error(globallogcat, error);
89  throw std::runtime_error{error};
90  }
91 
92  return result;
93  }
94 
95  // Makes a json rpc request with the given request value and (if successful) returns a
96  // std::pair<bool, Value>. Takes two arguments: the request, and a lambda that takes an rvalue
97  // response and returns an value (typically moved via something like `return
98  // std::move(response.whatever)`).
99  template <typename RPC, typename GetValue,
100  typename Value = decltype(std::declval<GetValue>()(typename RPC::response{}))>
101  std::pair<bool, Value> get_result_pair(const typename RPC::request& req, GetValue get_value, uint64_t* check_height = nullptr) const
102  {
103  std::pair<bool, Value> result;
104  auto& [success, value] = result;
105  success = false;
106 
107  if (m_offline)
108  return result;
109 
110  try {
111  value = get_value(invoke_json_rpc<RPC>(req));
112  success = true;
113  } catch (...) {}
114 
115  return result;
116  }
