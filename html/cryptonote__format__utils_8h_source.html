72  // skip_fields: How many fields of type <T> to skip
73  template<typename T>
74  bool find_tx_extra_field_by_type(const std::vector<tx_extra_field>& tx_extra_fields, T& field, size_t skip_fields = 0)
75  {
76  if (skip_fields >= tx_extra_fields.size())
77  return false;
78 
79  for (const auto& f : tx_extra_fields)
80  {
81  if (!std::holds_alternative<T>(f)) continue;
82  if (skip_fields == 0)
83  {
84  field = var::get<T>(f);
85  return true;
86  }
87  skip_fields--;
88  }
89 
90  return false;
91  }

97  // Adds some data to tx_extra with a tag looked up according to the given template parameter S. T
98  // does not have to be the same as S, but it must have the same size and must be a basic layout,
99  // no-padding type. Intended for use with simple tx extra decorator types, for example:
100  //
101  // add_tx_extra<tx_extra_pub_key>(tx.extra, pubkey);
102  //
103  template <typename S, typename T>
104  void add_tx_extra(std::vector<uint8_t>& tx_extra, const T& val)
105  {
106  static_assert(sizeof(S) == sizeof(T));
107  (void) tools::template_index<S, tx_extra_field>;
108 
109  add_tagged_data_to_tx_extra(tx_extra, serialization::variant_serialization_tag<S, uint8_t>, tools::view_guts(val));
110  }

227  // Returns code strings for various tx verification failures:
228  // - "failed" -- general "bad transaction" code
229  // - "altchain" -- the transaction is spending outputs that exist on an altchain.
230  // - "mixin" -- the transaction has the wrong number of decoys
231  // - "double_spend" -- the transaction is spending outputs that are already spent
232  // - "invalid_input" -- one or more inputs in the transaction are invalid
233  // - "invalid_output" -- out or more outputs in the transaction are invalid
234  // - "too_few_outputs" -- the transaction does not create enough outputs (at least two are
235  // required, currently).
236  // - "too_big" -- the transaction is too large
237  // - "overspend" -- the transaction spends (via outputs + fees) more than the inputs
238  // - "fee_too_low" -- the transaction fee is insufficient
239  // - "invalid_version" -- the transaction version is invalid (the wallet likely needs an update).
240  // - "invalid_type" -- the transaction type is invalid
241  // - "snode_locked" -- one or more outputs are currently staked to a registred service node and
242  // thus are not currently spendable on the blockchain.
243  // - "blacklisted" -- the outputs are currently blacklisted (from being in the 30-day penalty
244  // period following a service node deregistration).
245  // - "not_relayed" -- the transaction cannot be relayed to the network for some reason but may
246  // still have been accepted by this node.
