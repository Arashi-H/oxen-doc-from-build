133  struct nested_array {
134  json_archiver& ar;
135  int exc_count = std::uncaught_exceptions();
136  bool first = true;
137 
138  // Call before writing an element to add a delimiter. The first element() call adds no
139  // delimiter. Returns the archive itself, allowing you to write:
140  //
141  // auto arr = ar.begin_array();
142  // for (auto& val : whatever)
143  // value(arr.element(), val);
144  //
145  json_archiver& element() {
146  if (first) first = false;
147  else ar.delimit_array();
148  return ar;
149  }
150 
151  ~nested_array() noexcept(false) {
152  if (std::uncaught_exceptions() == exc_count) { // Normal destruction
153  --ar.depth_;
154  if (ar.inner_array_contents_)
155  ar.make_indent();
156  ar.stream_ << ']';
157  }
158  // else we're destructing during a stack unwind so some other serialization failed, thus don't
159  // try terminating the array (since it might *also* throw if an IO error occurs).
160  }
