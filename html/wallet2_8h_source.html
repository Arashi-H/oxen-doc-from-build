200  const crypto::hash &genesis() const { return m_genesis; }
201  void push_back(const crypto::hash &hash) { if (m_offset == 0 && m_blockchain.empty()) m_genesis = hash; m_blockchain.push_back(hash); }
202  bool is_in_bounds(size_t idx) const { return idx >= m_offset && idx < size(); }
203  const crypto::hash &operator[](size_t idx) const { return m_blockchain[idx - m_offset]; }
204  crypto::hash &operator[](size_t idx) { return m_blockchain[idx - m_offset]; }
205  void crop(size_t height) { m_blockchain.resize(height - m_offset); }
206  void clear() { m_offset = 0; m_blockchain.clear(); }
207  bool empty() const { return m_blockchain.empty() && m_offset == 0; }
208  void trim(size_t height) { while (height > m_offset && m_blockchain.size() > 1) { m_blockchain.pop_front(); ++m_offset; } m_blockchain.shrink_to_fit(); }
209  void refill(const crypto::hash &hash) { m_blockchain.push_back(hash); --m_offset; }

1304  bool invoke_http(const typename RPC::request& req, typename RPC::response& res, bool throw_on_error = false)
1305  {
1306  using namespace cryptonote::rpc;
1307  static_assert(std::is_base_of_v<RPC_COMMAND, RPC>
1308 #ifdef ENABLE_LIGHT_WALLET
1309  || std::is_base_of_v<tools::light_rpc::LIGHT_RPC_COMMAND, RPC>
1310 #endif
1311  );
1312 
1313  if (m_offline) return false;
1314 
1315  try {
1316  if constexpr (std::is_base_of_v<LEGACY, RPC>)
1317  // TODO: post-8.x hard fork we can remove this one and let everything go through the
1318  // non-binary json_rpc version instead (because all legacy json commands are callable via
1319  // json_rpc as of daemon 8.x).
1320  res = m_http_client.json<RPC>(RPC::names().front(), req);
1321  else if constexpr (std::is_base_of_v<BINARY, RPC>)
1322  res = m_http_client.binary<RPC>(RPC::names().front(), req);
1323  else if constexpr (std::is_base_of_v<RPC_COMMAND, RPC>)
1324  res = m_http_client.json_rpc<RPC>(RPC::names().front(), req);
1325  else // light RPC:
1326  res = m_http_client.json<RPC>(RPC::name, req);
1327  return true;
1328  } catch (const std::exception& e) {
1329  if (throw_on_error)
1330  throw;
1331  else
1332  log::error(log::Cat("wallet.wallet2"), "HTTP request failed: {}", e.what());
1333  } catch (...) {
1334  if (throw_on_error)
1335  throw;
1336  else
1337  log::error(log::Cat("wallet.wallet2"), "HTTP request failed: unknown error");
1338  }
