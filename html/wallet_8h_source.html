56 // Wrapper that holds a lock to prevent background refreshes, which kill things; provides `->`
57 // indirection into the tools::wallet2 instance.
58 struct LockedWallet {
59  std::unique_lock<std::recursive_timed_mutex> refresh_lock;
60  tools::wallet2* const wallet;
61  // Constructs a wallet wrapper from a moved existing unique_lock which may be initially locked
62  // or unlocked (if unlocked, it will be immediately locked).
63  LockedWallet(const std::unique_ptr<tools::wallet2>& w, std::unique_lock<std::recursive_timed_mutex>&& lock)
64  : refresh_lock{std::move(lock)}, wallet{w.get()} {
65  if (!refresh_lock) refresh_lock.lock();
66  }
67  // Constructs a wallet wrapper from a wallet and the refresh mutex; locks the mutex immediately.
68  LockedWallet(const std::unique_ptr<tools::wallet2>& w, std::recursive_timed_mutex& refresh_mutex)
69  : refresh_lock{refresh_mutex}, wallet{w.get()} {}
70 
71  // Returns the wallet2 pointer, to allow `w->whatever()` to call into wallet functions through
72  // the locking wrapper.
73  tools::wallet2* operator->() { return wallet; }
74 };
