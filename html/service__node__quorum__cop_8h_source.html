64  struct quorum_manager
65  {
66  std::shared_ptr<const quorum> obligations;
67  // TODO(doyle): Workers aren't used, but I kept this as a quorum
68  // to avoid drastic changes for now to a lot of the service node API
69  std::shared_ptr<const quorum> checkpointing;
70  std::shared_ptr<const quorum> blink;
71  std::shared_ptr<const quorum> pulse;
72 
73  std::shared_ptr<const quorum> get(quorum_type type) const
74  {
75  if (type == quorum_type::obligations) return obligations;
76  else if (type == quorum_type::checkpointing) return checkpointing;
77  else if (type == quorum_type::blink) return blink;
78  else if (type == quorum_type::pulse) return pulse;
79  log::error(log::Cat("quorum_cop"), "Developer error: Unhandled quorum enum with value: {}", (size_t)type);
80  assert(!"Developer error: Unhandled quorum enum with value: ");
81  return nullptr;
82  }
83  };
