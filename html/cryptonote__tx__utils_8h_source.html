242  struct oxen_construct_tx_params
243  {
244  hf hf_version = hf::hf7;
245  txtype tx_type = txtype::standard;
246 
247  // Can be set to non-zero values to have the tx be constructed specifying required burn amounts
248  // Note that the percentage is relative to the minimal base tx fee, *not* the actual tx fee.
249  //
250  // For example if the base tx fee is 0.5, the priority sets the fee to 500%, the fixed burn
251  // amount is 0.1, and the percentage burn is 300% then the tx overall fee will be 0.1+2.5=2.6,
252  // and the burn amount will be 0.1+3(0.5)=1.6 (and thus the miner tx coinbase amount will be
253  // 1.0). (See also wallet2's get_fee_percent which needs to return a value large enough to
254  // allow these amounts to be burned).
255  uint64_t burn_fixed = 0; // atomic units
256  uint64_t burn_percent = 0; // 123 = 1.23x base fee.
257  };
