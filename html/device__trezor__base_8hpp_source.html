126  template<class t_message=google::protobuf::Message>
127  std::shared_ptr<t_message>
128  client_exchange(const std::shared_ptr<const google::protobuf::Message> &req,
129  const std::optional<messages::MessageType> & resp_type = std::nullopt,
130  const std::optional<std::vector<messages::MessageType>> & resp_types = std::nullopt,
131  const std::optional<messages::MessageType*> & resp_type_ptr = std::nullopt,
132  bool open_session = false)
133  {
134  // Require strictly protocol buffers response in the template.
135  static_assert(std::is_base_of_v<google::protobuf::Message, t_message>);
136  const bool accepting_base = boost::is_same<google::protobuf::Message, t_message>::value;
137  if (resp_types && !accepting_base){
138  throw std::invalid_argument("Cannot specify list of accepted types and not using generic response");
139  }
140 
141  // Determine type of expected message response
142  const messages::MessageType required_type =
143  accepting_base ? messages::MessageType_Success :
144  resp_type ? *resp_type :
145  MessageMapper::get_message_wire_number<t_message>();
146 
147  // Open session if required
148  if (open_session){
149  try {
150  m_transport->open();
151  } catch (const std::exception& e) {
152  std::throw_with_nested(exc::SessionException("Could not open session"));
153  }
154  }
155 
156  // Scoped session closer
157  OXEN_DEFER {
158  if (open_session){
159  this->get_transport()->close();
160  }
161  };
162 
163  // Write/read the request
164  CHECK_AND_ASSERT_THROW_MES(req, "Request is null");
165  auto msg_resp = call_raw(req.get());
166 
167  bool processed = false;
168  do {
169  processed = message_handler(msg_resp);
170  } while(processed);
171 
172  // Response section
173  if (resp_type_ptr){
174  **resp_type_ptr = msg_resp.m_type;
175  }
176 
177  if (msg_resp.m_type == messages::MessageType_Failure) {
178  throw_failure_exception(dynamic_cast<messages::common::Failure *>(msg_resp.m_msg.get()));
179 
180  } else if (!accepting_base && msg_resp.m_type == required_type) {
181  return message_ptr_retype<t_message>(msg_resp.m_msg);
182 
183  } else if (accepting_base && (!resp_types ||
184  std::find(resp_types->begin(), resp_types->end(), msg_resp.m_type) != resp_types->end())) {
185  return message_ptr_retype<t_message>(msg_resp.m_msg);
186 
187  } else {
188  throw exc::UnexpectedMessageException(msg_resp.m_type, msg_resp.m_msg);
189  }
190  }
