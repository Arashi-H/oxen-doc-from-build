8  // Calculate the median element (the middle element, if an odd size, and the mean of the two
9  // middle elements if even). Pass first=true if you don't care about the mean of the middle two,
10  // in which case you'll get back the value of lower of the two middle elements.
11  // This leaves the given range in an indeterminant (partially sorted) order.
12  template <typename RandomAccessIter>
13  auto median(RandomAccessIter begin, RandomAccessIter end, bool first=false) {
14  std::size_t size = end - begin;
15  if (size == 0)
16  return std::decay_t<decltype(*begin)>{};
17 
18  auto mid = begin + (size - 1) / 2;
19  std::nth_element(begin, mid, end);
20 
21  if (first || size % 2)
22  return *mid;
23 
24  auto mid2 = std::min_element(mid + 1, end);
25  return (*mid + *mid2) / 2;
26  }
