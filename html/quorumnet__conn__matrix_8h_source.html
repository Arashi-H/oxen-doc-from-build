40 class quorum_conn_iterator {
41 private:
42  friend class quorum_outgoing_conns;
43  friend class quorum_incoming_conns;
44  const bool * const flags;
45  int i;
46  const int N;
47  const int step;
48 
49  quorum_conn_iterator(const bool *flags_, int N_, bool outgoing_ = true) : flags{flags_}, i{0}, N{N_}, step{outgoing_ ? 1 : N} {
50  if (flags == nullptr && N != 0)
51  throw std::domain_error("Invalid/unsupported quorum size (" + std::to_string(N) + ")");
52  if (!flags[0] && N > 0) ++*this;
53  }
54 
55 public:
56  using iterator_category = std::forward_iterator_tag;
57  using value_type = int;
58  using difference_type = int;
59  using pointer_type = const int *;
60  using reference = const int &;
61 
62  constexpr quorum_conn_iterator() : flags{nullptr}, i{0}, N{0}, step{1} {}
63 
64  reference operator*() const { return i; }
65 
66  bool operator==(const quorum_conn_iterator &it) const {
67  return (i == N && it.i == it.N) || (flags == it.flags && i == it.i);
68  }
69 
70  bool operator!=(const quorum_conn_iterator &it) const { return !(*this == it); }
71 
72  quorum_conn_iterator &operator++() {
73  do { ++i; } while (i < N && !flags[i*step]);
74  return *this;
75  }
76 
77  quorum_conn_iterator operator++(int) {
78  quorum_conn_iterator old{*this};
79  ++*this;
80  return old;
81  }
82 };
