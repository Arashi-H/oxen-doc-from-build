245 // Serializes some value with a predicate that must be satisfied when deserializing. If the
246 // predicate fails the value serialization raises an exception. The predicate is invoked (during
247 // deserialization) with a reference to `v` (which has already been updated).
248 template <class Archive, typename T, typename Predicate>
249 void value(Archive& ar, T& v, Predicate test)

251  value(ar, v);

265 void varint(Archive& ar, T& val)
266 {
267  using UType = std::underlying_type_t<T>;
268  UType tmp;
269  if (Archive::is_serializer)
270  tmp = static_cast<UType>(val);

272  varint(ar, tmp);
