369  void update(uint64_t height, bool force = false)
370  {
371  auto current_time = std::chrono::system_clock::now();
372  const auto node_update_threshold = TARGET_BLOCK_TIME / 2;
373  if (node_update_threshold < current_time - m_blockchain_height_update_time || m_blockchain_height <= height)
374  {
375  update_blockchain_height();
376  m_blockchain_height = (std::max)(m_blockchain_height, height);
377  }
378 
379  if (std::chrono::milliseconds(20) < current_time - m_print_time || force)
380  {
381  std::cout << QT_TRANSLATE_NOOP("cryptonote::simple_wallet", "Height ") << height << " / " << m_blockchain_height << '\r' << std::flush;
382  m_print_time = current_time;
383  }
384  }
385 
386  private:
387  void update_blockchain_height()
388  {
389  std::string err;
390  uint64_t blockchain_height = m_simple_wallet.get_daemon_blockchain_height(err);
391  if (err.empty())
392  {
393  m_blockchain_height = blockchain_height;
394  m_blockchain_height_update_time = std::chrono::system_clock::now();
395  }
396  else
397  {
398  log::error(log::Cat("wallet.simplewallet"), "Failed to get current blockchain height: {}", err);
399  }
400  }
