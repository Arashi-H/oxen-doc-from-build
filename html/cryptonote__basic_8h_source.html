304  if (!signatures_expected)

316  value(arr.element(), signatures[i]);
317  }
318  }

333  {

337  vin.size() > 0 && std::holds_alternative<txin_to_key>(vin[0]) ? var::get<txin_to_key>(vin[0]).key_offsets.size() - 1 : 0);
338  }
339  }
340  }

342  pruned = false;
