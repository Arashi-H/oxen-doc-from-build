60  template<typename T>
61  inline T utf8prefix(const T &s, size_t count)
62  {
63  T prefix = "";
64  size_t avail = s.size();
65  const char *ptr = s.data();
66  while (count-- && avail--)
67  {
68  prefix += *ptr++;
69  while (avail && ((*ptr) & 0xc0) == 0x80)
70  {
71  prefix += *ptr++;
72  --avail;
73  }
74  }
75  return prefix;
76  }

78  template<typename T>
79  inline T utf8canonical(const T &s)
80  {
81  T sc = "";
82  size_t avail = s.size();
83  const char *ptr = s.data();
84  wint_t cp = 0;
85  int bytes = 1;
86  char wbuf[8], *wptr;
87  while (avail--)
88  {
89  if ((*ptr & 0x80) == 0)
90  {
91  cp = *ptr++;
92  bytes = 1;
93  }
94  else if ((*ptr & 0xe0) == 0xc0)
95  {
96  if (avail < 1)
97  throw std::runtime_error("Invalid UTF-8");
98  cp = (*ptr++ & 0x1f) << 6;
99  cp |= *ptr++ & 0x3f;
100  --avail;
101  bytes = 2;
102  }
103  else if ((*ptr & 0xf0) == 0xe0)
104  {
105  if (avail < 2)
106  throw std::runtime_error("Invalid UTF-8");
107  cp = (*ptr++ & 0xf) << 12;
108  cp |= (*ptr++ & 0x3f) << 6;
109  cp |= *ptr++ & 0x3f;
110  avail -= 2;
111  bytes = 3;
112  }
113  else if ((*ptr & 0xf8) == 0xf0)
114  {
115  if (avail < 3)
116  throw std::runtime_error("Invalid UTF-8");
117  cp = (*ptr++ & 0x7) << 18;
118  cp |= (*ptr++ & 0x3f) << 12;
119  cp |= (*ptr++ & 0x3f) << 6;
120  cp |= *ptr++ & 0x3f;
121  avail -= 3;
122  bytes = 4;
123  }
124  else
125  throw std::runtime_error("Invalid UTF-8");
126 
127  cp = std::towlower(cp);
128  wptr = wbuf;
129  switch (bytes)
130  {
131  case 1: *wptr++ = cp; break;
132  case 2: *wptr++ = 0xc0 | (cp >> 6); *wptr++ = 0x80 | (cp & 0x3f); break;
133  case 3: *wptr++ = 0xe0 | (cp >> 12); *wptr++ = 0x80 | ((cp >> 6) & 0x3f); *wptr++ = 0x80 | (cp & 0x3f); break;
134  case 4: *wptr++ = 0xf0 | (cp >> 18); *wptr += 0x80 | ((cp >> 12) & 0x3f); *wptr++ = 0x80 | ((cp >> 6) & 0x3f); *wptr++ = 0x80 | (cp & 0x3f); break;
135  default: throw std::runtime_error("Invalid UTF-8");
136  }
137  *wptr = 0;
138  sc += T(wbuf, bytes);
139  cp = 0;
140  bytes = 1;
141  }
142  return sc;
143  }

188  {
189  int ii;
190  std::vector<std::string>::const_iterator it;
191  if (word_list.size () != NWORDS)
192  throw std::runtime_error("Wrong word list length for " + language_name);
193  for (it = word_list.begin(), ii = 0; it != word_list.end(); it++, ii++)
194  {
195  word_map[*it] = ii;
196  if ((*it).size() < unique_prefix_length)
197  {
198  if (flags & ALLOW_SHORT_WORDS)
199  log::debug(globallogcat, "{} word '{}' is shorter than its prefix length, {}", language_name, *it, unique_prefix_length);
200  else
201  throw std::runtime_error("Too short word in " + language_name + " word list: " + *it);
202  }
203  epee::wipeable_string trimmed;
204  if (it->length() > unique_prefix_length)
205  {
206  trimmed = utf8prefix(*it, unique_prefix_length);
207  }
208  else
209  {
210  trimmed = *it;
211  }
212  if (trimmed_word_map.find(trimmed) != trimmed_word_map.end())
213  {
214  if (flags & ALLOW_DUPLICATE_PREFIXES)
215  log::warning(globallogcat, "Duplicate prefix in {} word list: {}", language_name, std::string(trimmed.data(), trimmed.size()));
216  else
217  throw std::runtime_error("Duplicate prefix in " + language_name + " word list: " + std::string(trimmed.data(), trimmed.size()));
218  }
219  trimmed_word_map[trimmed] = ii;
220  }
