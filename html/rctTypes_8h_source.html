286  template <typename Archive>
287  void serialize_rctsig_base(Archive &ar, size_t inputs, size_t outputs)
288  {
289  field_varint(ar, "type", type, [](const RCTType& x) { return x <= RCTType::CLSAG; });
290  if (type == RCTType::Null)
291  return;
292  if (!tools::equals_any(type, RCTType::Full, RCTType::Simple, RCTType::Bulletproof, RCTType::Bulletproof2, RCTType::CLSAG))
293  throw std::invalid_argument{"invalid ringct type"};
294 
295  field_varint(ar, "txnFee", txnFee);
296 
297  // inputs/outputs not saved, only here for serialization help
298  // FIELD(message) - not serialized, it can be reconstructed
299  // FIELD(mixRing) - not serialized, it can be reconstructed
300  if (type == RCTType::Simple) // moved to prunable with bulletproofs
301  {
302  auto arr = start_array(ar, "pseudoOuts", pseudoOuts, inputs);
303  for (auto& e : pseudoOuts)
304  value(arr.element(), e);
305  }
306 
307  {
308  auto arr = start_array(ar, "ecdhInfo", ecdhInfo, outputs);
309  if (tools::equals_any(type, RCTType::Bulletproof2, RCTType::CLSAG))
310  {
311  for (auto& e : ecdhInfo) {
312  auto obj = arr.element().begin_object();
313  if (Archive::is_deserializer)
314  memset(e.amount.bytes, 0, sizeof(e.amount.bytes));
315  field(ar, "amount", reinterpret_cast<crypto::hash8&>(e.amount));
316  }
317  } else {
318  for (auto& e : ecdhInfo)
319  value(arr.element(), e);
320  }
321  }
322 
323  {
324  auto arr = start_array(ar, "outPk", outPk, outputs);
325  for (auto& e : outPk)
326  value(arr.element(), e.mask);
327  }
328  }

337  // when changing this function, update cryptonote::get_pruned_transaction_weight
338  template<typename Archive>
339  void serialize_rctsig_prunable(Archive &ar, RCTType type, size_t inputs, size_t outputs, size_t mixin)
340  {
341  if (type == RCTType::Null)
342  return;
343  if (!tools::equals_any(type, RCTType::Full, RCTType::Simple, RCTType::Bulletproof, RCTType::Bulletproof2, RCTType::CLSAG))
344  throw std::invalid_argument{"invalid ringct type"};
345  if (rct::is_rct_bulletproof(type))
346  {
347  uint32_t nbp = bulletproofs.size();
348  if (tools::equals_any(type, RCTType::Bulletproof2, RCTType::CLSAG))
349  field_varint(ar, "nbp", nbp);
350  else
351  field(ar, "nbp", nbp);
352  if (nbp > outputs)
353  throw std::invalid_argument{"too many bulletproofs"};
354 
355  auto arr = start_array(ar, "bp", bulletproofs, nbp);
356  for (auto& b : bulletproofs)
357  value(arr.element(), b);
358 
359  if (auto n_max = n_bulletproof_max_amounts(bulletproofs); n_max < outputs)
360  throw std::invalid_argument{"invalid bulletproofs: n_max (" + std::to_string(n_max) + ") < outputs (" + std::to_string(outputs) + ")"};
361  }
362  else
363  {
364  auto arr = start_array(ar, "rangeSigs", rangeSigs, outputs);
365  for (auto& s : rangeSigs)
366  value(arr.element(), s);
367  }
368 
369  if (type == RCTType::CLSAG)
370  {
371  auto arr = start_array(ar, "CLSAGs", CLSAGs, inputs);
372 
373  for (auto& clsag : CLSAGs)
374  {
375  // we save the CLSAGs contents directly, because we want it to save its
376  // arrays without the size prefixes, and the load can't know what size
377  // to expect if it's not in the data
378  auto obj = arr.element().begin_object();
379  {
380  auto arr_s = start_array(ar, "s", clsag.s, mixin + 1);
381  for (auto& x : clsag.s)
382  value(arr_s.element(), x);
383  }
384  field(ar, "c1", clsag.c1);
385  field(ar, "D", clsag.D);
386  }
387  }
388  else
389  {
390  // we keep a byte for size of MGs, because we don't know whether this is
391  // a simple or full rct signature, and it's starting to annoy the hell out of me
392 
393  size_t mg_elements = 1, mg_ss2_elements = inputs + 1;
394  if (tools::equals_any(type, RCTType::Simple, RCTType::Bulletproof, RCTType::Bulletproof2)) {
395  mg_elements = inputs;
396  mg_ss2_elements = 2;
397  }
398 
399  {
400  auto arr = start_array(ar, "MGs", MGs, mg_elements);
401  for (auto& mg : MGs)
402  {
403  auto obj = arr.element().begin_object();
404 
405  // we save the MGs contents directly, because we want it to save its
406  // arrays and matrices without the size prefixes, and the load can't
407  // know what size to expect if it's not in the data
408  {
409  auto arr_ss = start_array(ar, "ss", mg.ss, mixin + 1);
410  for (auto& ss : mg.ss)
411  {
412  auto arr_ss2 = arr_ss.element().begin_array();
413  if constexpr (Archive::is_deserializer)
414  ss.resize(mg_ss2_elements);
415  else if (ss.size() != mg_ss2_elements)
416  throw std::invalid_argument{"invalid mg_ss2 size: have " + std::to_string(ss.size()) + ", expected " + std::to_string(mg_ss2_elements)};
417 
418  for (auto& x : ss)
419  value(arr_ss2.element(), x);
420  }
421  }
422  field(ar, "cc", mg.cc);
423  // MGs[i].II not saved, it can be reconstructed
424  }
425  }
426  }
427  if (tools::equals_any(type, RCTType::Bulletproof, RCTType::Bulletproof2, RCTType::CLSAG))
428  {
429  auto arr = start_array(ar, "pseudoOuts", pseudoOuts, inputs);
430  for (auto& o : pseudoOuts)
431  value(arr.element(), o);
432  }
433  }
