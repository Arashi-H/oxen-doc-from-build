78 void serialize(Archive &a, wallet::tx_construction_data &x, const unsigned int ver)
79 {
80  a & x.sources;
81  a & x.change_dts;
82  a & x.splitted_dsts;
83  if (ver < 2)
84  {
85  // load list to vector
86  std::list<size_t> selected_transfers;
87  a & selected_transfers;
88  x.selected_transfers.clear();
89  x.selected_transfers.reserve(selected_transfers.size());
90  for (size_t t: selected_transfers)
91  x.selected_transfers.push_back(t);
92  }
93  a & x.extra;
94  a & x.unlock_time;
95  a & x.dests;
96  if (ver < 1)
97  {
98  x.subaddr_account = 0;
99  return;
100  }
101  a & x.subaddr_account;
102  a & x.subaddr_indices;
103  if (!typename Archive::is_saving())
104  {
105  x.rct_config = { rct::RangeProofType::Borromean, 0 };
106  if (ver < 6)
107  {
108  x.tx_type = cryptonote::txtype::standard;
109  x.hf_version = cryptonote::hf::hf13_enforce_checkpoints;
110  }
111  }
112 
113  if (ver < 2)
114  return;
115  a & x.selected_transfers;
116  if (ver < 3)
117  return;
118  if (ver < 5)
119  {
120  bool use_bulletproofs = x.rct_config.range_proof_type != rct::RangeProofType::Borromean;
121  a & use_bulletproofs;
122  if (!typename Archive::is_saving())
123  x.rct_config = { use_bulletproofs ? rct::RangeProofType::Bulletproof : rct::RangeProofType::Borromean, 0 };
124  return;
125  }
126  a & x.rct_config;
127 
128  if (ver < 6) return;
129  a & x.tx_type;
130  a & x.hf_version;
131 }
