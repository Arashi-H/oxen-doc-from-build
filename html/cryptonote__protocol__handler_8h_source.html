134  template<class T>
135  bool relay_to_synchronized_peers(typename T::request& arg, cryptonote_connection_context& exclude_context)
136  {
137  log::debug(globallogcat, "[{}] post relay {} -->", epee::net_utils::print_connection_context_short(exclude_context), tools::type_name<T>());
138  std::vector<std::pair<epee::net_utils::zone, boost::uuids::uuid>> connections;
139  m_p2p->for_each_connection([&exclude_context, &connections](connection_context& context, nodetool::peerid_type peer_id)
140  {
141  if (context.m_state > cryptonote_connection_context::state_synchronizing)
142  {
143  epee::net_utils::zone zone = context.m_remote_address.get_zone();
144  if (peer_id && exclude_context.m_connection_id != context.m_connection_id)
145  connections.push_back({zone, context.m_connection_id});
146  }
147  return true;
148  });
149 
150  if (connections.size())
151  {
152  std::string arg_buff;
153  epee::serialization::store_t_to_binary(arg, arg_buff);
154  return m_p2p->relay_notify_to_list(T::ID, epee::strspan<uint8_t>(arg_buff), std::move(connections));
155  }
156 
157  return true;
158  }
