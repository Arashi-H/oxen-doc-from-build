50  // Returns the height range for which the given block/network version is valid. Returns a pair of
51  // heights {A, B} where A/B is the first/last height at which the version is acceptable. Returns
52  // nullopt for A if the version indicates a hardfork we do not know about (i.e. we are likely
53  // outdated), and returns nullopt for B if the version indicates that top network version we know
54  // about (i.e. there is no subsequent hardfork scheduled).
55  std::pair<std::optional<uint64_t>, std::optional<uint64_t>>
56  get_hard_fork_heights(network_type type, hf version);

58  // Returns the lowest network version >= the given version, that is, it rounds up missing hf table
59  // entries to the next largest entry. Typically this returns the network version itself, but if
60  // some versions are skipped (particularly on testnet/devnet/fakechain) then this will return the
61  // next version that does exist in the hard fork list. If there is no >= value in the hard fork
62  // table then this returns the given hard fork value itself.
63  //
64  // For example, if the HF list contains hf versions {7,8,14} then:
65  // hard_fork_ceil(7) == 7
66  // hard_fork_ceil(8) == 8
67  // hard_fork_ceil(9) == 14
68  // ...
69  // hard_fork_ceil(14) == 14
70  // hard_fork_ceil(15) == 15
71  hf hard_fork_ceil(network_type type, hf version);

74  // version.
75  bool is_hard_fork_at_least(network_type type, hf version, uint64_t height);

81  // Returns the network (i.e. block) version for the given height.
82  inline hf get_network_version(network_type nettype, uint64_t height) {
83  return get_network_version_revision(nettype, height).first;
84  }
85 
86  // Returns the first height at which the given network version rules become active. This is
87  // a shortcut for `get_hard_fork_heights(type, hard_fork_ceil(type, version)).first`, i.e. it
88  // returns the first height at which `version` rules become active (even if they became active at
89  // a hard fork > the given value).
90  inline std::optional<uint64_t> hard_fork_begins(network_type type, hf version) {
91  return get_hard_fork_heights(type, hard_fork_ceil(type, version)).first;
92  }

94  // Returns the "ideal" network version that we want to use on blocks we create, which is to use
95  // the required major version and current minor version. (Minor versions are sometimes used to
96  // change network features, but do not change the blockchain rules).
97  // Before HF19, the minor version must be >= the major version, and is set to the largest major
98  // version we know about.
99  std::pair<hf, uint8_t> get_ideal_block_version(network_type nettype, uint64_t height);
