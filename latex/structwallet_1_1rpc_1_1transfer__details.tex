\doxysubsection{Detailed Description}
Transfer details struct


\begin{DoxyItemize}
\item {\ttfamily amount} -- Amount of this transfer.
\item {\ttfamily spent} -- Indicates if this transfer has been spent.
\item {\ttfamily global\+\_\+index} -- The index into the global list of transactions grouped by amount in the Loki network.
\item {\ttfamily tx\+\_\+hash} -- Several incoming transfers may share the same hash if they were in the same transaction.
\item {\ttfamily subaddr\+\_\+index} -- Major \& minor index, account and subaddress index respectively.
\item {\ttfamily key\+\_\+image} -- Key image for the incoming transfer\textquotesingle{}s unspent output (empty unless verbose is true).
\item {\ttfamily block\+\_\+height} -- \mbox{\hyperlink{structwallet_1_1Block}{Block}} height the transfer occurred on
\item {\ttfamily frozen} -- If the output has been intentionally frozen by the user, i.\+e. unspendable.
\item {\ttfamily unlocked} -- If the TX is spendable yet 
\end{DoxyItemize}
