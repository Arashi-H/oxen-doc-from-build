\doxysubsection{Detailed Description}
Payment details struct


\begin{DoxyItemize}
\item {\ttfamily payment\+\_\+id} -- Payment ID matching the input parameter.
\item {\ttfamily tx\+\_\+hash} -- Transaction hash used as the transaction ID.
\item {\ttfamily amount} -- Amount for this payment.
\item {\ttfamily block\+\_\+height} -- Height of the block that first confirmed this payment.
\item {\ttfamily unlock\+\_\+time} -- Time (in block height) until this payment is safe to spend.
\item {\ttfamily locked} -- If the payment is spendable or not
\item {\ttfamily subaddr\+\_\+index} -- Major \& minor index, account and subaddress index respectively.
\item {\ttfamily address} -- Address receiving the payment.
